\def\changecatcodes{%
    % Tunes, start of a string line
    \catcode`C=13
    \catcode`D=13
    \catcode`E=13
    \catcode`F=13
    \catcode`G=13
    \catcode`A=13
    \catcode`H=13
    %
    \catcode`X=13 % to show chords above
    %
    \catcode`\|=11
    \catcode`\~=11
    \catcode`\^^M=13 % newline
    \catcode`-=12
}

\gdef\tune#1{\raisebox{1pt}{\hbox to 10pt{\scriptsize#1\hss}}}
\gdef\notune#1{\raisebox{1pt}{\hbox to 10pt{}}}
