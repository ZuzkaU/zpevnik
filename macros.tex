\newcommand{\songTitle}[1]{
  {\Huge\sffamily\bfseries\MakeUppercase{#1}}
}

\newcommand{\songAuthor}[1]{
  {\LARGE\sffamily\slshape #1}
}

\newcommand{\mysong}[3]{
  \clearpage
  \songTitle{#1}\\[8pt]
  \songAuthor{#2}\\[8pt]
  \begin{guitarMagic}
  \begin{guitarCr}
  #3
  \end{guitarCr}
  \end{guitarMagic}
}

\newenvironment{song}[3]{
  \clearpage
  \centering
  \songTitle{#1}\\[8pt]
  \songAuthor{#2}\\[8pt]
  \vskip -30pt
  \begin{guitar}
  \leftskip=#3
  \Large
}{  
  \end{guitar}
}

\renewcommand{\verse}[1]{
  \vskip 10pt
  \hskip -23pt
  {\LARGE\bfseries #1.}
  \hskip 3pt
}

\newcommand{\chorus}{
  \vskip 10pt
  \hskip -28pt
  {\LARGE\bfseries R:}
  \hskip 3pt
}

\renewcommand{\guitarPreAccord}{\sffamily\bfseries\Large}

\renewcommand\guitarEndLine{\medskip\par\leavevmode}
\renewcommand\guitarEndPar{\par\leavevmode}

  \linepenalty=50\relax\parskip0pt\relax}
